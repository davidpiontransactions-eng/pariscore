// server.js — predictCorners()
function predictCorners(homeAvg, awayAvg, thresholds = [6.5, 7.5, 8.5, 9.5, 10.5], sampleSize = null) {
  const expectedTotal = (homeAvg + awayAvg) / 2;
  const lambda = expectedTotal;
  const probs = {};
  for (const threshold of thresholds) {
    let cumulative = 0;
    const maxX = Math.floor(threshold);
    for (let k = 0; k <= maxX; k++) {
      let logP = -lambda + k * Math.log(Math.max(lambda, 0.001));
      for (let i = 1; i <= k; i++) logP -= Math.log(i);
      cumulative += Math.exp(logP);
    }
    probs[`over_${String(threshold).replace(''.'', ''_'')}`] = Math.round((1 - cumulative) * 100);
  }
  return {
    expected_total: Math.round(lambda * 10) / 10,
    probabilities: probs,
    confidence: sampleSize == null ? 25 : Math.min(85, Math.round(35 + 50 * (sampleSize / (sampleSize + 6)))),
  };
}

// server.js — cornerProbFromXG() (frontend fallback)
function cornerProbFromXG(homeXG, awayXG) {
  const lambda = Math.max(1.0, (homeXG + awayXG) * 3.0);
  let cumulative = 0;
  for (let k = 0; k <= 6; k++) {
    let logP = -lambda + k * Math.log(Math.max(lambda, 0.001));
    for (let i = 1; i <= k; i++) logP -= Math.log(i);
    cumulative += Math.exp(logP);
  }
  return Math.round(Math.max(0, Math.min(99, (1 - cumulative) * 100)));
}

// server.js — enrichCornersFromBSD()
async function enrichCornersFromBSD(matches) {
  for (const m of matches) {
    const homeData = await fetchBSDTeamCornerHistory(m.home_team, bsdLeagueId, 5);
    const awayData = await fetchBSDTeamCornerHistory(m.away_team, bsdLeagueId, 5);
    const homeAvg = homeData?.totalCornersPerMatch ?? 5.25;
    const awayAvg = awayData?.totalCornersPerMatch ?? 5.25;
    const prediction = predictCorners(homeAvg, awayAvg, undefined, sampleSize);
    m.corners_poisson = { over_6_5: prediction.probabilities.over_6_5, source: ''bsd'', home_avg: homeAvg, away_avg: awayAvg };
  }
}

==============================================================
FORMULE MATEMATICHE DA APPLICARE
==============================================================

Per OGNI mercato statistico (shots, sot, cards, fouls, offsides, saves):

1. Recupera media storica per squadra (ultimi N match):
   avg_home = media(shots_home_ultimi_N_match)
   avg_away = media(shots_away_ultimi_N_match)

2. Calcola λ atteso per la partita:
   λ = (avg_home + avg_away) / 2

3. Distribuzione Poisson:
   P(X = k) = λ^k × e^(-λ) / k!

4. Probabilità Over X.5:
   P(Over X.5) = 1 - Σ_{k=0}^{floor(X)} P(X = k)

5. Cote teorica equa:
   cote = 1 / P(Over X.5)

6. Edge vs bookmaker reale:
   Edge = (P(Over X.5) - 1/cote_bookmaker) × 100
